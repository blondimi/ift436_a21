\section*{Question 2}

\begin{enumerate}[(a)]

\item Algorithme de tri rapide:

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \SetKwFunction{trier}{trier}
    \SetKwFunction{trierp}{trier'}
    \SetKwFunction{partition}{partition}
    \SetKwProg{myproc}{}{}{}
    
    \KwIn{séquence $s$ d'éléments comparables}
    \Sortie{séquence $s$ triée}
    \algskip
    
    \myproc{\trier{$s$}\algdelim}{
      \myproc(\tcp*[f]{partionne $s$ autour de $x = s[hi]$ et}){%
        \partition{$lo, hi$}\algdelim}{%

        $x \leftarrow s[hi];\ i \leftarrow lo$
        \tcp*{retourne le nouvel index $i$ de $x$}
        \algskip
        
        \For{$j \leftarrow lo, \ldots, hi$}{
          \If{$s[j] < x$}{
            $s[i] \leftrightarrow s[j]$\;
            $i \leftarrow i + 1$\;
          }
        }
        
        \algskip
        $s[i] \leftrightarrow s[hi]$\;
        
        \algskip
        \Return{$i$}
      }
      \algskip
      \algskip
      \algskip
      
      \myproc{\trierp{$lo, hi$}\algdelim}{
        \If{$lo < hi$}{
          $pivot \leftarrow \partition{$lo, hi$}$\;
          \trierp{$lo, pivot - 1$}\tcp*{Trier le côté gauche}
          \trierp{$pivot + 1, hi$}\tcp*{Trier le côté droit}
        }
      }
      \algskip
      \algskip
      \algskip
      
      \trierp{$1, |s|$}
      \algskip
      
      \Return{$s$}
    }
  \end{algorithm}
